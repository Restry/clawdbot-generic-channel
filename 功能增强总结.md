## 使用示例

### 1. 添加表情反应

```typescript
import { addReaction } from "@restry/generic-channel";

const reaction = addReaction({
  messageId: "msg-123",
  chatId: "chat-456",
  senderId: "user-789",
  emoji: "👍",
});
```

### 2. 编辑消息

```typescript
import { editMessage } from "@restry/generic-channel";

const edit = editMessage({
  messageId: "msg-123",
  chatId: "chat-456",
  senderId: "user-789",
  newContent: "更新后的消息",
});
```

### 3. 搜索消息

```typescript
import { searchMessages } from "@restry/generic-channel";

const results = searchMessages({
  query: "重要文档",
  chatId: "chat-456",
  limit: 20,
});
```

### 4. 群组管理

```typescript
import { createGroup, addGroupMember } from "@restry/generic-channel";

const group = createGroup({
  groupId: "group-123",
  groupName: "项目团队",
  createdBy: "user-789",
});

addGroupMember({
  groupId: "group-123",
  userId: "user-abc",
  invitedBy: "user-789",
});
```

### 5. 消息置顶

```typescript
import { pinMessage } from "@restry/generic-channel";

const pinned = pinMessage({
  messageId: "msg-123",
  chatId: "chat-456",
  pinnedBy: "user-789",
  expiresAt: Date.now() + 86400000, // 24小时后自动取消
});
```
